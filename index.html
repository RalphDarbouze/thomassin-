const https = require('https'); function checkServer() { const options = { hostname: 'your-server-url.com', port: 443, path: '/api/stats', method: 'GET', timeout: 5000 }; const req = https.request(options, (res) => { console.log(`✅ Server is UP - Status: ${res.statusCode}`); }); req.on('error', (error) => { console.error(`❌ Server is DOWN - Error: ${error.message}`); }); req.on('timeout', () => { console.error('❌ Server timeout'); req.destroy(); }); req.end(); } // Check every 5 minutes setInterval(checkServer, 5 * 60 * 1000); checkServer();
